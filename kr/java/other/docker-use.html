# Docker CP 多容器文件分发

基于 docker cp 命令实现本地 Linux 向多容器进行文件分发复制

#!/bin/bash

# docker-cp
# 该脚本是基于 docker cp 命令实现本地 Linux 向多个容器指定目录分发复制本地 Linux 目录文件
# 第一个参数：本地目录文件路径
# 第二个参数：容器目录文件路径
# 第三个以上参数：容器名称

# 参数判断：至少三个参数
if [ $# -lt 3 ];then
        echo Error: Not Enough Arguement!
        exit;
fi

localfile=$1
targetfile=$2

# 判断本地文件或目录是否存在
if [ ! -e $localfile ];then
        echo Error: $localfile: No such file or directory
        exit;
fi

shift 2         # 跳过前面 2 个参数
# 遍历容器，分发文件
for container in $@
do
        if [ "$(docker container ls -a -q -f name=$container 2> /dev/null)" == "" ]; then
                # 容器不存在
                echo Error: $container: no such container
                continue
        elif [ $(docker inspect --format '{{.State.Running}}' $container) == false ]; then
                # 容器停止状态
                echo Error: $container: is stop
                continue
        fi
        cpflag=true
        existflag=$(docker exec -it $container sh -c 'if [ -e '$targetfile' ]; then echo exist; fi')
        fileflag=$(docker exec -it $container sh -c 'if [ -f '$targetfile' ]; then echo exist; fi')
        driflag=$(docker exec -it $container sh -c 'if [ -d '$targetfile' ]; then echo exist; fi')
        if [[ $targetfile = */ && ! $existflag ]];then
                # 以 / 结尾，所指容器路径为目录，且该目录不存在，则创建
                echo $container: $targetfile: No such directory
                docker exec -it $container sh -c 'mkdir -p '$targetfile''
                echo $container: $targetfile: create this directory
        fi
        if [[ $targetfile != */ && $existflag && $fileflag ]];then
                # 不以 / 结尾，且已经存在，且所指容器路径是个文件
                if [ -d $localfile ];then
                        echo $container error: $localfile is a directory, $targetfile is a file
                        cpflag=false
                fi
        fi
        if [[ $targetfile != */ && ! $existflag ]];then
                # 不以 / 结尾，可能是文件或目录，且所指容器路径不存在
                # 那么只需要判断容器所指路径最后第二级是否存在，不存在直接创建即可
                if [ ! -e $(dirname $targetfile) ];then
                        echo $container: $(dirname $targetfile): No such directory
                        docker exec -it $container sh -c 'mkdir -p $(dirname '$targetfile')'
                        echo $container: $(dirname $targetfile): create directory
                fi
        fi
        if [ $cpflag = true ];then
                docker cp $localfile $container:$targetfile
                echo Success: $container: $localfile '->' $targetfile, successfully finished
        fi
done

# 远程连接 Docker Engine

Docker 默认只接受本地客户端的请求，远程连接 Docker Engine 有两种连接方式：

  • TCP：在服务端开始 TCP 连接，此连接为不安全连接，数据通过明文进行传输，缺省端口 2375
  • TCP_TLS：在 TCP 基础上加了 SSL 的安全证书，以保证连接安全，缺省端口 2376

# TCP 无认证连接

[xzx@xzx ~]$ vim /usr/lib/systemd/system/docker.service             # 配置远程客户端连接 Docker Engine
# 在该行最后增加 -H tcp://0.0.0.0:2375，表示不限制访问地址，启动 TCP 连接
ExecStart=/usr/bin/dockerd -H fd:// --containerd=/run/containerd/containerd.sock -H tcp://0.0.0.0:2375

[xzx@xzx ~]$ systemctl daemon-reload                            # 重载配置
[xzx@xzx ~]$ systemctl restart docker                           # 重启 docker 服务
[xzx@xzx ~]$ netstat -tnpl | grep 2375                          # 查看监听端口
[xzx@xzx ~]$ firewall-cmd --zone=public --add-port=2375/tcp --permanent     # 开放防火墙端口，permanent：表示持久
[xzx@xzx ~]$ firewall-cmd --zone=public --remove-port=2375/tcp --permanent     # 关闭防火墙端口，permanent：表示持久
[xzx@xzx ~]$ firewall-cmd --reload                              # 重启防火墙
[xzx@xzx ~]$ docker -H tcp://192.168.120.159:2375 version       # 远程连接 Docker 并执行命令 version

# 容器时间修改

宿主机使用 CST(China Standard Time) 而 Docker 容器使用 UTC(Coordinated Universal Time)，那么两者就会相差 8 小时，修改容器时间方式：

export TZ=Asia/Shanghai  # 直接进入容器修改时区
# 更新软链接
[root@dmcdw-3 etc]# date
Sun Sep 26 22:09:10 UTC 2021
[root@dmcdw-3 etc]# ll /etc/localtime
lrwxrwxrwx. 1 root root 25 Nov 13  2020 /etc/localtime -> ../usr/share/zoneinfo/UTC
[root@dmcdw-3 etc]# ln -snf /usr/share/zoneinfo/Asia/Shanghai /etc/localtime        # 更新软连接
[root@dmcdw-3 etc]# ll /etc/localtime
lrwxrwxrwx. 1 root root 33 Sep 27 06:02 /etc/localtime -> /usr/share/zoneinfo/Asia/Shanghai
[root@dmcdw-3 etc]# date
Mon Sep 27 06:07:24 CST 2021
# docker run 创建容器时挂载宿主机 localtime 文件到容器内
docker run --name <name> -v /etc/localtime:/etc/localtime:ro
