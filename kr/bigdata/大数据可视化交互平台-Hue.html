# 大数据可视化交互平台 Hue

Hue Guide (opens new window)

HUE 是一个开源 Apache Hadoop UI 系统，HUE 是基于 Python Web 框架 Django 实现；通过使用 HUE 工具，可以使用浏览器控制台对 Hadoop 生态圈中的各个组件进行交互控制。

# 镜像安装
docker search hue           # 搜索镜像
docker pull gethue/hue      # 拉取镜像
# 启动容器，开启服务
docker run -itd --restart=no --net=dmcdw-bridge --privileged -p 8888:8888 --name hue --hostname hue gethue/hue
docker exec -it --user root hue bash        # 以 root 身份进入容器

Hue 元数据库是 Sqlite，容易产生 database is locked 异常，将元数据库改为 MySQL：

# 创建数据库、用户及授权
mysql> create database hue;
mysql> create user 'xzx'@'%' identified by 'xzx';
mysql> grant all privileges on *.* to 'xzx'@'%';                   # 授权
mysql> flush privileges;                                           # 刷新权限
mysql> show grants for 'xzx'@'%';                                  # 查询用户权限
# 初始化数据库
/usr/share/hue/build/env# bin/hue syncdb
/usr/share/hue/build/env# bin/hue migrate

[desktop]
    time_zone=Asia/Shanghai
    [[database]]                # Hue 元数据库改为 MySQL，解决 database is lock 问题

        host=192.168.72.1
        port=3306

        password=hue

[[interpreters]]
    [[[mysql]]]                 # 集成 MySQL
        name = MySQL
        interface=sqlalchemy
        options='{"url": "mysql://hue:hue@192.168.72.1:3306"}'
    [[[sqoop1]]]
        name=Sqoop1
        interface=oozie
    [[[hive]]]
        name=Hive
        interface=hiveserver2
    [[[shell]]]
        name=Shell
        interface=oozie
[beeswax]
    hive_server_host=xzx-hive
    hive_server_port=10000
[hadoop]                        # Hadoop 集群配置
    [[hdfs_clusters]]
        [[[default]]]
            fs_defaultfs=hdfs://xzx-hadoop-master:9000
            webhdfs_url=http://xzx-hadoop-master:9870/webhdfs/v1
    [[yarn_clusters]]
        [[[default]]]
            resourcemanager_host=xzx-hadoop-master
            resourcemanager_api_url=http://xzx-hadoop-master:8088
[zookeeper]
    [[clusters]]
        [[[default]]]
            host_ports=xzx-zk-1:2181,xzx-zk-2:2181,xzx-zk-3:2181
